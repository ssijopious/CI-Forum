PROCEDURE "PROC_USERTYPES_INSUPDTE" 
(
  P_USER_TYPE_ID IN NUMBER 
, P_USER_TYPE IN VARCHAR2 
, P_STATUS IN NUMBER 
, P_ID IN NUMBER 
, P_RET OUT VARCHAR2
) AS 
O_RET NUMBER;
BEGIN
IF P_USER_TYPE_ID = 0 THEN  
    SELECT SEQ_USER_TYPES.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_USER_TYPE_ID;
END IF;
  MERGE INTO USERTYPES X
USING (SELECT P_USER_TYPE_ID as USER_TYPE_ID FROM dual) Y
ON (X.USER_TYPE_ID = Y.USER_TYPE_ID)
    WHEN MATCHED THEN
        UPDATE SET X.USER_TYPE = P_USER_TYPE, 
                   X.STATUS = P_STATUS, 	
                   X.MODIFIED_BY =	P_ID,
                   X.MODIFIED_DATE = TO_DATE(SYSDATE,'dd-mon-rrrr') 	
        WHERE X.USER_TYPE_ID = Y.USER_TYPE_ID
    WHEN NOT MATCHED THEN
        INSERT (X.USER_TYPE_ID ,X.USER_TYPE,X.STATUS,X.CREATED_BY,X.CREATED_DATE)
        values(O_RET, P_USER_TYPE,P_STATUS,P_ID,TO_DATE(SYSDATE,'dd-mon-rrrr'));
        
        --
        SELECT O_RET INTO O_RET from dual;
      dbms_output.put_line(sql%rowcount ||','|| P_RET);
END;
