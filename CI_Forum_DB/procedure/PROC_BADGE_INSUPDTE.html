PROCEDURE "PROC_BADGE_INSUPDTE" 
(
P_BADGE_ID 			  IN NUMBER
,P_BADGE_TYPE_ID 	IN NUMBER
,P_USER_ID 			  IN NUMBER
,P_STATUS         IN NUMBER 
,P_ID             IN NUMBER 
,P_DATE           IN DATE 
,P_RET            out NUMBER
) AS 
O_RET NUMBER;
BEGIN

IF P_BADGE_ID = 0 THEN  
    SELECT SEQ_BADGES.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_BADGE_ID;
END IF;

  MERGE INTO BADGES X
USING (SELECT P_BADGE_ID as BADGE_ID FROM dual) Y
ON (X.BADGE_ID = Y.BADGE_ID)
    WHEN MATCHED THEN
        UPDATE SET X.BADGE_TYPE_ID 	= P_BADGE_TYPE_ID
                  ,X.USER_ID 			  = P_USER_ID 
                  ,X.STATUS         = P_STATUS 	
                  ,X.MODIFIED_BY    =	P_ID
                  ,X.MODIFIED_DATE  = P_DATE 
        WHERE X.BADGE_ID = Y.BADGE_ID
    WHEN NOT MATCHED THEN    
        INSERT (X.BADGE_ID ,X.BADGE_TYPE_ID ,X.USER_ID ,X.STATUS ,X.CREATED_BY ,X.CREATED_DATE)
        VALUES (O_RET ,P_BADGE_TYPE_ID ,P_USER_ID ,P_STATUS ,P_ID ,P_DATE);
        
        
        SELECT 0 INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
