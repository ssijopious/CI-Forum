PROCEDURE "PROC_USER_DATA_SELECT" 
(
  P_USER_ID    IN NUMBER
, P_CUR      out SYS_REFCURSOR
) AS 

BEGIN

OPEN P_CUR FOR
SELECT US.USER_ID
,US.USER_CODE 
,US.NAME1
,US.NAME2
,US.USER_TYPE AS USER_TYPE_ID
,UT.USER_TYPE AS USER_TYPE
,US.CREATED_DATE
,US.DESCRIPTION
,US.AGE
,US.DISPLAY_NAME
,US.WEBSITE_URL 
,LC.USERNAME
,LC.LOGIN_PASSWORD
,US.LAST_ACCESS_DATE
,NVL(VTS.TOTAL_BOUNTY , 1) TOTAL_BOUNTY 
FROM USERSDETAILS US
JOIN USERTYPES UT ON UT.USER_TYPE_ID = US.USER_TYPE
LEFT JOIN LOGINCREDENTIALS LC ON LC.USER_ID=US.USER_ID
LEFT JOIN (SELECT VS.USER_ID, SUM(VS.TOTAL_BOUNTY_AMOUNT) TOTAL_BOUNTY FROM VOTETOTAL VS GROUP BY VS.USER_ID) VTS ON VTS.USER_ID = US.USER_ID
WHERE US.USER_ID=P_USER_ID;
    
END;
