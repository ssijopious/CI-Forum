PROCEDURE "PROC_POSTS_SELECT" 
(
  P_POST_ID  IN NUMBER
, P_USER_ID    IN NUMBER
, P_CUR      out SYS_REFCURSOR
) AS 

BEGIN

OPEN P_CUR FOR
SELECT POST_ID
      ,POST_TYPE_ID
      ,PARENT_POST_ID
      ,USER_ID
      ,TITLE
      ,TAG_TYPE_ID
      ,SUB_TAGS
      ,BODY_CONTENT
      ,CREATED_DATE
      ,VIEW_COUNT
      ,LAST_ACTIVITY_DATE
      ,LAST_EDITED_DATE
      ,LAST_EDITOR_USER_ID
      ,LAST_EDITOR_DISPLAY_NAME
      ,FAVORITE_COUNT
      ,ACCEPTED_ANSWER_ID
      ,ANSWER_COUNT
      ,CLOSED_DATE
      ,COMMENTCOUNT
      ,CLOSED_BY
FROM  POSTS
WHERE  POST_ID = COALESCE(NULLIF(P_POST_ID, NULL), POST_ID)
AND    USER_ID = COALESCE(NULLIF(P_USER_ID, NULL), USER_ID)
    ORDER BY POST_ID;
    
END;
