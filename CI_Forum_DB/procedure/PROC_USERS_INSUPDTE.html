PROCEDURE "PROC_USERS_INSUPDTE" 
(
P_USER_ID			         IN NUMBER
,P_USER_CODE           IN VARCHAR2
,P_NAME1               IN VARCHAR2
,P_NAME2               IN VARCHAR2
,P_USER_TYPE           IN NUMBER
,P_STATUS              IN NUMBER
,P_MODIFIED_BY         IN NUMBER
,P_DESCRIPTION         IN NVARCHAR2
,P_AGE                 IN NUMBER
,P_DISPLAY_NAME        IN VARCHAR2
,P_LAST_ACCESS_DATE    IN DATE
,P_REPUTATION          IN NUMBER
,P_VIEWERS             IN NUMBER
,P_WEBSITE_URL         IN VARCHAR2
,P_PARENT_ID           IN NUMBER
,P_RET             out NUMBER
) AS 
O_RET NUMBER;
BEGIN

IF P_USER_ID = 0 THEN  
    SELECT SEQ_USERS.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_USER_ID;
END IF;

  MERGE INTO USERSDetails X
USING (SELECT P_USER_ID as USER_ID FROM dual) Y
ON (X.USER_ID = Y.USER_ID)
    WHEN MATCHED THEN
        UPDATE SET X.USER_CODE         = COALESCE(NULLIF(P_USER_CODE, NULL), X.USER_CODE)         
                  ,X.NAME1             = COALESCE(NULLIF(P_NAME1, NULL), X.NAME1)             
                  ,X.NAME2             = COALESCE(NULLIF(P_NAME2, NULL), X.NAME2)             
                  ,X.USER_TYPE         = COALESCE(NULLIF(P_USER_TYPE, NULL), X.USER_TYPE)         
                  ,X.STATUS            = COALESCE(NULLIF(P_STATUS, NULL), X.STATUS)    
                  ,X.MODIFIED_BY       = COALESCE(NULLIF(P_MODIFIED_BY, NULL), X.MODIFIED_BY)           
                  ,X.MODIFIED_DATE     = TO_DATE(SYSDATE,'dd-mon-rrrr')     
                  ,X.DESCRIPTION       = COALESCE(NULLIF(P_DESCRIPTION, NULL), X.DESCRIPTION)       
                  ,X.AGE               = COALESCE(NULLIF(P_AGE, NULL), X.AGE)               
                  ,X.DISPLAY_NAME      = COALESCE(NULLIF(P_DISPLAY_NAME, NULL), X.DISPLAY_NAME)      
                  ,X.LAST_ACCESS_DATE  = COALESCE(NULLIF(P_LAST_ACCESS_DATE, NULL), TO_DATE(SYSDATE,'dd-mon-rrrr'))  
                  ,X.REPUTATION        = COALESCE(NULLIF(P_REPUTATION, NULL), X.REPUTATION)        
                  ,X.VIEWERS           = COALESCE(NULLIF(P_VIEWERS, NULL), X.VIEWERS)             
                  ,X.WEBSITE_URL       = COALESCE(NULLIF(P_WEBSITE_URL, NULL), X.WEBSITE_URL)       
                  ,X.PARENT_ID         = COALESCE(NULLIF(P_PARENT_ID, NULL), X.PARENT_ID)   
        WHERE X.USER_ID = Y.USER_ID
    WHEN NOT MATCHED THEN    
        INSERT ( X.USER_ID ,X.USER_CODE ,X.NAME1 ,X.NAME2 ,X.USER_TYPE ,X.STATUS ,X.CREATED_DATE ,X.MODIFIED_BY ,X.MODIFIED_DATE ,X.DESCRIPTION 
                ,X.AGE ,X.DISPLAY_NAME ,X.LAST_ACCESS_DATE ,X.REPUTATION ,X.VIEWERS ,X.WEBSITE_URL ,X.PARENT_ID )
        VALUES ( O_RET ,P_USER_CODE ,P_NAME1 ,P_NAME2 ,P_USER_TYPE ,P_STATUS ,TO_DATE(SYSDATE,'dd-mon-rrrr') ,NULL ,NULL ,P_DESCRIPTION      
                ,P_AGE ,P_DISPLAY_NAME ,TO_DATE(SYSDATE,'dd-mon-rrrr') ,P_REPUTATION ,P_VIEWERS ,P_WEBSITE_URL ,P_PARENT_ID);
        
        
        SELECT O_RET INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
