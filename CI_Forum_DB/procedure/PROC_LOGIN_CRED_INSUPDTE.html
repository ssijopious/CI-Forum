PROCEDURE "PROC_LOGIN_CRED_INSUPDTE" 
(
P_LOGIN_CREDENTIAL_ID 	IN NUMBER
,P_USER_ID 				      IN NUMBER
,P_USERNAME 			      IN VARCHAR2
,P_LOGIN_PASSWORD	      IN VARCHAR2
,P_VALID_TILL 		      IN DATE 
,P_STATUS               IN NUMBER 
,P_ID                   IN NUMBER 
,P_RET                  out NUMBER
) AS 
O_RET NUMBER;
BEGIN

IF P_LOGIN_CREDENTIAL_ID = 0 THEN  
    SELECT SEQ_LOGIN_CREDENTIALS.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_LOGIN_CREDENTIAL_ID;
END IF;

  MERGE INTO LOGINCREDENTIALS X
USING (SELECT P_LOGIN_CREDENTIAL_ID as LOGIN_CREDENTIAL_ID FROM dual) Y
ON (X.LOGIN_CREDENTIAL_ID = Y.LOGIN_CREDENTIAL_ID)
    WHEN MATCHED THEN
        UPDATE SET X.USER_ID 				= P_USER_ID 				
                  ,X.USERNAME 			= COALESCE(NULLIF(P_USERNAME, NULL), X.USERNAME) 			
                  ,X.LOGIN_PASSWORD = COALESCE(NULLIF(P_LOGIN_PASSWORD, NULL), X.LOGIN_PASSWORD) 			
                  ,X.VALID_TILL 		= COALESCE(NULLIF(P_VALID_TILL, NULL), TO_DATE(SYSDATE,'dd-mon-rrrr'))   
                  ,X.STATUS         = P_STATUS 	
                  ,X.MODIFIED_BY    =	P_ID
                  ,X.MODIFIED_DATE  = TO_DATE(SYSDATE,'dd-mon-rrrr') 
        WHERE X.LOGIN_CREDENTIAL_ID = Y.LOGIN_CREDENTIAL_ID
    WHEN NOT MATCHED THEN    
        INSERT (X.LOGIN_CREDENTIAL_ID ,X.USER_ID ,X.USERNAME ,X.LOGIN_PASSWORD ,X.VALID_TILL ,X.STATUS ,X.CREATED_BY ,X.CREATED_DATE)
        VALUES (O_RET ,P_USER_ID ,P_USERNAME ,P_LOGIN_PASSWORD ,P_VALID_TILL ,P_STATUS ,P_ID ,TO_DATE(SYSDATE,'dd-mon-rrrr'));
        
        
         SELECT O_RET INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
