PROCEDURE "PROC_BADGETYPES_INSUPDTE" 
(
 P_BADGE_TYPE_ID IN NUMBER 
,P_BADGE_TYPE IN VARCHAR2
,P_STATUS IN NUMBER 
,P_ID IN NUMBER 
,P_DATE IN DATE 
,P_RET out number
) AS 
O_RET NUMBER;
BEGIN

IF P_BADGE_TYPE_ID = 0 THEN  
    SELECT SEQ_BADGE_TYPES.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_BADGE_TYPE_ID;
END IF;
  MERGE INTO BADGETYPES X
USING (SELECT P_BADGE_TYPE_ID as BADGE_TYPE_ID FROM dual) Y
ON (X.BADGE_TYPE_ID = Y.BADGE_TYPE_ID)
    WHEN MATCHED THEN
        UPDATE SET X.BADGE_TYPE = P_BADGE_TYPE, 
                   X.STATUS = P_STATUS, 	
                   X.MODIFIED_BY =	P_ID,
                   X.MODIFIED_DATE = P_DATE 	
        WHERE X.BADGE_TYPE_ID = Y.BADGE_TYPE_ID
    WHEN NOT MATCHED THEN
        INSERT (x.BADGE_TYPE_ID ,X.BADGE_TYPE ,X.STATUS ,X.CREATED_BY ,X.CREATED_DATE)
        values(O_RET ,P_BADGE_TYPE ,P_STATUS ,P_ID ,P_DATE);
        
         SELECT 0 INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
