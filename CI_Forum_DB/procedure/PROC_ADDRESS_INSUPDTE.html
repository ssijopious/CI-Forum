PROCEDURE "PROC_ADDRESS_INSUPDTE" 
(
P_ADDRESS_ID    IN NUMBER 
,P_USER_ID      IN NUMBER 
,P_LINE1        IN VARCHAR2
,P_LINE2        IN VARCHAR2
,P_DISTRICT     IN VARCHAR2
,P_STATES       IN VARCHAR2
,P_COUNTRY      IN VARCHAR2
,P_POST_CODE    IN VARCHAR2
,P_MASTER_TYPE  IN NUMBER 
,P_AFFILIATION  IN VARCHAR2
,P_STATUS       IN NUMBER 
,P_ID           IN NUMBER 
,P_RET          out NUMBER
) AS 
O_RET NUMBER;
BEGIN

IF P_ADDRESS_ID = 0 THEN  
    SELECT seq_address.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET :=P_ADDRESS_ID;
END IF;

  MERGE INTO ADDRESS X
USING (SELECT P_ADDRESS_ID as ADDRESS_ID FROM dual) Y
ON (X.ADDRESS_ID = Y.ADDRESS_ID)
    WHEN MATCHED THEN
        UPDATE SET X.LINE1 = COALESCE(NULLIF(P_LINE1, NULL), X.LINE1)      
                  ,X.LINE2 = P_LINE2      
                  ,X.DISTRICT = P_DISTRICT   
                  ,X.STATES = P_STATES      
                  ,X.COUNTRY = P_COUNTRY    
                  ,X.POST_CODE = P_POST_CODE  
                  ,X.MASTER_TYPE = P_MASTER_TYPE
                  ,X.AFFILIATION = P_AFFILIATION
                  ,X.STATUS = P_STATUS 	
                  ,X.MODIFIED_BY =	P_ID
                  ,X.MODIFIED_DATE = TO_DATE(SYSDATE,'dd-mon-rrrr') 
        WHERE X.ADDRESS_ID = Y.ADDRESS_ID
    WHEN NOT MATCHED THEN    
        INSERT (X.ADDRESS_ID ,X.USER_ID ,X.LINE1 ,X.LINE2 ,X.DISTRICT ,X.STATES ,X.COUNTRY ,X.POST_CODE ,X.MASTER_TYPE ,X.AFFILIATION ,X.STATUS,X.CREATED_BY,X.CREATED_DATE)
        VALUES (O_RET ,P_USER_ID ,P_LINE1 ,P_LINE2 ,P_DISTRICT ,P_STATES ,P_COUNTRY ,P_POST_CODE ,P_MASTER_TYPE ,P_AFFILIATION ,P_STATUS,P_ID,TO_DATE(SYSDATE,'dd-mon-rrrr'));
        
        
        SELECT O_RET INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
