PROCEDURE "PROC_VOTES_INSUPDTE" 
(
P_VOTED_BY        IN NUMBER 
,P_VOTED_TO        IN NUMBER 
,P_VOTE_TYPE_ID    IN NUMBER 
,P_POST_ID         IN NUMBER 
,P_TAG_TYPE_ID     IN NUMBER 
,P_STATUS          IN NUMBER 
,P_ID              IN NUMBER
,P_RET             out NUMBER
) AS 
O_BOUNTY_AMOUNT    NUMBER; 
O_TOTAL_BOUNTY_AMOUNT NUMBER;

BEGIN

IF (P_VOTED_BY <> P_VOTED_TO) THEN

BEGIN
      SELECT NVL(TOTAL_BOUNTY_AMOUNT, 1)/NVL(TOTAL_VOTES,1) 
      INTO O_BOUNTY_AMOUNT 
      FROM VOTETOTAL
      WHERE USER_ID=P_VOTED_BY
      AND TAG_TYPE_ID = P_TAG_TYPE_ID;
EXCEPTION
      WHEN NO_DATA_FOUND THEN
        O_BOUNTY_AMOUNT := 1;
END;

IF(O_BOUNTY_AMOUNT<=1) THEN 
    O_BOUNTY_AMOUNT := 1;
END IF;

BEGIN
      SELECT NVL(TOTAL_BOUNTY_AMOUNT, 1)--/NVL(TOTAL_VOTES,1) 
      INTO O_TOTAL_BOUNTY_AMOUNT 
      FROM VOTETOTAL
      WHERE USER_ID=P_VOTED_TO
      AND TAG_TYPE_ID = P_TAG_TYPE_ID;
EXCEPTION
      WHEN NO_DATA_FOUND THEN
        O_TOTAL_BOUNTY_AMOUNT := 1;
END;

IF(P_VOTE_TYPE_ID = 1) THEN 
    O_TOTAL_BOUNTY_AMOUNT := O_TOTAL_BOUNTY_AMOUNT + O_BOUNTY_AMOUNT;
ELSE
    O_TOTAL_BOUNTY_AMOUNT := O_TOTAL_BOUNTY_AMOUNT - O_BOUNTY_AMOUNT;
END IF;

IF(O_TOTAL_BOUNTY_AMOUNT<=0) THEN 
    O_TOTAL_BOUNTY_AMOUNT := 1;
END IF;

  MERGE INTO VOTES X
USING (SELECT P_POST_ID AS POST_ID, P_VOTED_BY as VOTED_BY FROM dual) Y --(SELECT 0 AS POST_ID, 0 as VOTED_BY FROM dual) Y--
ON (X.POST_ID||X.VOTED_BY = Y.POST_ID||Y.VOTED_BY )

    WHEN MATCHED THEN
        UPDATE SET X.BOUNTY_AMOUNT  = NVL(O_BOUNTY_AMOUNT,0.001)
                  ,X.VOTED_TO       = P_VOTED_TO     
                  ,X.VOTE_TYPE_ID   = P_VOTE_TYPE_ID 
                  ,X.TAG_TYPE_ID    = P_TAG_TYPE_ID 
                  ,X.STATUS         = P_STATUS 	
                  ,X.MODIFIED_BY    =	P_ID
                  ,X.MODIFIED_DATE  = TO_DATE(SYSDATE,'dd-mon-rrrr') 
        WHERE X.POST_ID = Y.POST_ID
        AND X.VOTED_BY = Y.VOTED_BY
        
    WHEN NOT MATCHED THEN    
        INSERT (X.BOUNTY_AMOUNT ,X.VOTED_BY ,X.VOTED_TO ,X.VOTE_TYPE_ID ,X.POST_ID ,X.TAG_TYPE_ID ,X.STATUS  ,X.CREATED_DATE)
        VALUES (O_BOUNTY_AMOUNT ,P_VOTED_BY ,P_VOTED_TO ,P_VOTE_TYPE_ID ,P_POST_ID ,P_TAG_TYPE_ID ,P_STATUS ,TO_DATE(SYSDATE,'dd-mon-rrrr'));
        

   MERGE INTO VOTETOTAL A
USING (SELECT P_TAG_TYPE_ID AS TAG_TYPE_ID, P_VOTED_TO as USER_ID FROM dual) B
ON (A.USER_ID||A.TAG_TYPE_ID = B.USER_ID||B.TAG_TYPE_ID )

    WHEN MATCHED THEN
        UPDATE SET A.TOTAL_BOUNTY_AMOUNT  = O_TOTAL_BOUNTY_AMOUNT
                  ,A.TOTAL_VOTES       = A.TOTAL_VOTES+1 
        WHERE A.USER_ID = B.USER_ID
        AND A.TAG_TYPE_ID = B.TAG_TYPE_ID
        
    WHEN NOT MATCHED THEN    
        INSERT (A.USER_ID ,A.TAG_TYPE_ID ,A.TOTAL_BOUNTY_AMOUNT ,A.TOTAL_VOTES)
        VALUES (P_VOTED_TO ,P_TAG_TYPE_ID ,O_TOTAL_BOUNTY_AMOUNT ,1);
         SELECT 2 INTO P_RET from dual;        
ELSE
SELECT 0 INTO P_RET from dual;
END IF;       
      
END;
