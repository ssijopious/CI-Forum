PROCEDURE "PROC_POSTS_INSUPDTE" 
(
P_POST_ID					   IN NUMBER
,P_POST_TYPE_ID                IN NUMBER
,P_PARENT_POST_ID              IN NUMBER
,P_USER_ID                     IN NUMBER
,P_TITLE                       IN NVARCHAR2
,P_TAG_TYPE_ID                 IN NUMBER
,P_SUB_TAGS                    IN NVARCHAR2
,P_BODY_CONTENT                IN NVARCHAR2
--,P_CREATED_DATE                IN DATE
,P_VIEW_COUNT                  IN NUMBER
,P_LAST_ACTIVITY_DATE          IN DATE
,P_LAST_EDITED_DATE            IN DATE
,P_LAST_EDITOR_USER_ID         IN NUMBER
,P_LAST_EDITOR_DISPLAY_NAME    IN VARCHAR2
,P_FAVORITE_COUNT              IN NUMBER
,P_ACCEPTED_ANSWER_ID          IN NUMBER
,P_ANSWER_COUNT                IN NUMBER
,P_CLOSED_DATE                 IN DATE
,P_COMMENTCOUNT                IN NUMBER
,P_CLOSED_BY                   IN NUMBER
,P_FILE_NAME                IN NVARCHAR2
,P_SUB_PARENT_POST_ID          IN NUMBER
,P_RET             out NUMBER
) AS 
O_RET NUMBER;
O_PARENT_POST_ID NUMBER;
BEGIN

IF P_POST_ID = 0 THEN  
    SELECT SEQ_POSTS.NEXTVAL
    INTO   O_RET
    FROM   dual;
    ELSE
    O_RET := P_POST_ID;
END IF;
IF P_PARENT_POST_ID = 0 THEN
    O_PARENT_POST_ID := NULL;
ELSE
    O_PARENT_POST_ID := P_PARENT_POST_ID;
END IF;

  MERGE INTO POSTS X
USING (SELECT P_POST_ID as POST_ID FROM dual) Y
ON (X.POST_ID = Y.POST_ID)
    WHEN MATCHED THEN
        UPDATE SET X.POST_TYPE_ID               = P_POST_TYPE_ID               
                  ,X.PARENT_POST_ID             = O_PARENT_POST_ID             
                  ,X.USER_ID                    = P_USER_ID                    
                  ,X.TITLE                      = P_TITLE                      
                  ,X.TAG_TYPE_ID                = P_TAG_TYPE_ID                
                  ,X.SUB_TAGS                   = P_SUB_TAGS                   
                  ,X.BODY_CONTENT               = P_BODY_CONTENT               
                  ,X.VIEW_COUNT                 = P_VIEW_COUNT                 
                  ,X.LAST_ACTIVITY_DATE         = TO_DATE(SYSDATE,'dd-mon-rrrr')        
                  ,X.LAST_EDITED_DATE           = P_LAST_EDITED_DATE           
                  ,X.LAST_EDITOR_USER_ID        = P_LAST_EDITOR_USER_ID        
                  ,X.LAST_EDITOR_DISPLAY_NAME   = P_LAST_EDITOR_DISPLAY_NAME   
                  ,X.FAVORITE_COUNT             = P_FAVORITE_COUNT             
                  ,X.ACCEPTED_ANSWER_ID         = P_ACCEPTED_ANSWER_ID         
                  ,X.ANSWER_COUNT               = P_ANSWER_COUNT               
                  ,X.CLOSED_DATE                = P_CLOSED_DATE                
                  ,X.COMMENTCOUNT               = P_COMMENTCOUNT               
                  ,X.CLOSED_BY                  = P_CLOSED_BY   
                  ,X.FILE_NAME                  = P_FILE_NAME
                  ,X.SUB_PARENT_POST_ID         = P_SUB_PARENT_POST_ID
        WHERE X.POST_ID = Y.POST_ID
    WHEN NOT MATCHED THEN    
        INSERT ( X.POST_ID ,X.POST_TYPE_ID ,X.PARENT_POST_ID ,X.USER_ID ,X.TITLE ,X.TAG_TYPE_ID ,X.SUB_TAGS ,X.BODY_CONTENT ,X.CREATED_DATE ,X.VIEW_COUNT                 
                ,X.LAST_ACTIVITY_DATE ,X.LAST_EDITED_DATE ,X.LAST_EDITOR_USER_ID ,X.LAST_EDITOR_DISPLAY_NAME ,X.FAVORITE_COUNT ,X.ACCEPTED_ANSWER_ID         
                ,X.ANSWER_COUNT ,X.CLOSED_DATE ,X.COMMENTCOUNT ,X.CLOSED_BY ,X.FILE_NAME ,X.SUB_PARENT_POST_ID)
        VALUES ( O_RET ,P_POST_TYPE_ID ,O_PARENT_POST_ID ,P_USER_ID ,P_TITLE ,P_TAG_TYPE_ID ,P_SUB_TAGS ,P_BODY_CONTENT ,TO_DATE(SYSDATE,'dd-mon-rrrr') ,NULL                 
                ,TO_DATE(SYSDATE,'dd-mon-rrrr') ,TO_DATE(SYSDATE,'dd-mon-rrrr') ,P_LAST_EDITOR_USER_ID ,P_LAST_EDITOR_DISPLAY_NAME ,NULL ,NULL         
                ,NULL ,NULL ,NULL ,NULL ,P_FILE_NAME ,P_SUB_PARENT_POST_ID );
       
       IF P_PARENT_POST_ID <> 0 and P_POST_TYPE_ID = 2 THEN
        UPDATE POSTS SET 
                          COMMENTCOUNT               = NVL(COMMENTCOUNT,0) + 1
                        , LAST_EDITED_DATE           = TO_DATE(SYSDATE,'dd-mon-rrrr')
                        , LAST_ACTIVITY_DATE         = TO_DATE(SYSDATE,'dd-mon-rrrr')
                        , LAST_EDITOR_USER_ID        = P_LAST_EDITOR_USER_ID
                        , LAST_EDITOR_DISPLAY_NAME   = P_LAST_EDITOR_DISPLAY_NAME                        
        WHERE POST_ID=P_PARENT_POST_ID;
        END IF;
        
        IF P_PARENT_POST_ID <> 0 and P_POST_TYPE_ID = 3 THEN
        UPDATE POSTS SET 
                          ANSWER_COUNT        = NVL(ANSWER_COUNT,0) + 1                          
                        , LAST_ACTIVITY_DATE  = TO_DATE(SYSDATE,'dd-mon-rrrr')
        WHERE POST_ID=P_PARENT_POST_ID;
       END IF;
       
       SELECT O_RET INTO P_RET from dual;
      dbms_output.put_line(sql%rowcount || ',' || O_RET);
END;
